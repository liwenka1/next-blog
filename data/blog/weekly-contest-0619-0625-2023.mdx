[6892. 最大字符串配对数目](https://leetcode.cn/problems/find-maximum-number-of-string-pairs/)

```js
/**
 * @param {string[]} words
 * @return {number}
 */
var maximumNumberOfStringPairs = function (words) {
  const n = words.length
  let ans = 0
  for (let i = 0; i < n; i++) {
    const word = words[i]
    for (let j = i + 1; j < n; j++) {
      if (word == words[j].split('').reverse().join('')) ans++
    }
  }
  return ans
}
```

[6895. 构造最长的新字符串](https://leetcode.cn/problems/construct-the-longest-new-string/)

```js
/**
 * @param {number} x
 * @param {number} y
 * @param {number} z
 * @return {number}
 */
var longestString = function (x, y, z) {
  let ans = 0
  if (x != y) {
    ans += 2 * Math.min(x, y) + 1 + z
  } else {
    ans += x + y + z
  }
  return ans * 2
}
```

```js
/**
 * @param {string[]} words
 * @return {number}
 */
var minimizeConcatenatedLength = function (words) {
  const n = words.length
  const map = new Map()
  const dfs = (i, j, k) => {
    if (i == n) return 0
    const key = [i, j, k].join('-')
    if (map.has(key)) return map.get(key)
    const w = words[i]
    let res1 = dfs(i + 1, w[0], k) - (w[w.length - 1] == j ? 1 : 0)
    let res2 = dfs(i + 1, j, w[w.length - 1]) - (w[0] == k ? 1 : 0)
    map.set(key, Math.min(res1, res2) + w.length)
    return Math.min(res1, res2) + w.length
  }
  const w = words[0]
  return dfs(1, w[0], w[w.length - 1]) + w.length
}

    if (map.has(key)) return map.get(key)

  }
```

[2748. 美丽下标对的数目](https://leetcode.cn/problems/number-of-beautiful-pairs/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var countBeautifulPairs = function (nums) {
  const n = nums.length
  let ans = 0
  for (let i = 0; i < n; i++) {
    const str1 = nums[i].toString().split('')
    const n1 = str1[0]
    for (let j = i + 1; j < n; j++) {
      const str2 = nums[j].toString().split('')
      const n2 = str2[str2.length - 1]
      if (gcd(n1, n2) == 1) ans++
    }
  }
  return ans
}
const gcd = (a, b) => {
  return b == 0 ? a : gcd(b, a % b)
}
```

```js
/**

var makeTheIntegerZero = function (num1, num2) {
  for (let k = 1; k <= num1 - num2 * k; k++) {
    if (k >= (num1 - num2 * k).toString(2).split('1').length - 1) return k
  }
```

[6910. 将数组划分成若干好子数组的方式](https://leetcode.cn/problems/ways-to-split-array-into-good-subarrays/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var numberOfGoodSubarraySplits = function (nums) {
  const MOD = 1e9 + 7
  const n = nums.length
  let ans = 1
  let pre = -1
  for (let i = 0; i < n; i++) {
    if (nums[i] == 0) continue
    if (pre >= 0) ans = (ans * (i - pre)) % MOD
    pre = i
  }
  return pre == -1 ? 0 : ans
}
```
