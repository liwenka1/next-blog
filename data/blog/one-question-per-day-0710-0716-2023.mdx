[16. 最接近的三数之和](https://leetcode.cn/problems/3sum-closest/)

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var threeSumClosest = function (nums, target) {
  nums.sort((a, b) => a - b)
  let n = nums.length
  let ans1 = Infinity,
    ans2 = Infinity
  for (let i = 0; i < n - 2; i++) {
    if (i > 0 && nums[i] == nums[i - 1]) continue
    let left = i + 1
    let right = n - 1
    while (left < right) {
      let sum = nums[i] + nums[left] + nums[right] - target
      if (sum > 0) {
        right--
        ans1 = Math.min(ans1, sum)
      } else if (sum < 0) {
        left++
        ans2 = Math.min(ans2, -sum)
      } else {
        return target
      }
    }
  }
  return ans1 > ans2 ? -ans2 + target : ans1 + target
}
```

[1911. 最大子序列交替和](https://leetcode.cn/problems/maximum-alternating-subsequence-sum/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxAlternatingSum = function (nums) {
  let ans = 0
  const n = nums.length
  for (let i = 1; i < n; i++) {
    if (nums[i] < nums[i - 1]) {
      ans += nums[i - 1] - nums[i]
    }
  }
  return (ans += nums[n - 1])
}
```

[2544. 交替数字和](https://leetcode.cn/problems/alternating-digit-sum/)

```js
/**
 * @param {number} n
 * @return {number}
 */
var alternateDigitSum = function (n) {
  let ans = 0,
    sign = 1
  while (n > 0) {
    ans += (n % 10) * sign
    sign = -sign
    n = Math.floor(n / 10)
  }
  return ans * -sign
}
```

[931. 下降路径最小和](https://leetcode.cn/problems/minimum-falling-path-sum/)

```js
/**
 * @param {number[][]} matrix
 * @return {number}
 */
var minFallingPathSum = function (matrix) {
  const n = matrix.length
  const m = matrix[0].length
  const step = [-1, 0, 1]
  let dp = [...matrix[0]]
  for (let i = 1; i < n; i++) {
    let f = new Array(m).fill(Infinity)
    for (let j = 0; j < m; j++) {
      for (let x of step) {
        if (j + x >= 0 && j + x < m) {
          f[j] = Math.min(f[j], matrix[i][j] + dp[j + x])
        }
      }
    }
    dp = [...f]
  }
  return Math.min(...dp)
}
```

[979. 在二叉树中分配硬币](https://leetcode.cn/problems/distribute-coins-in-binary-tree/)

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var distributeCoins = function (root) {
  let ans = 0
  const dfs = (node) => {
    if (node == null) return 0
    const d = dfs(node.left) + dfs(node.right) + node.val - 1
    ans += Math.abs(d)
    return d
  }
  dfs(root)
  return ans
}
```

[18. 四数之和](https://leetcode.cn/problems/4sum/)

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[][]}
 */
var fourSum = function (nums, target) {
  nums.sort((a, b) => a - b)
  let n = nums.length
  if (n < 4) return []
  let ans = []
  for (let j = 0; j < n - 3; j++) {
    if (j > 0 && nums[j] == nums[j - 1]) continue
    if (nums[j] + nums[j + 1] + nums[j + 2] + nums[j + 3] > target) break
    if (nums[j] + nums[n - 1] + nums[n - 2] + nums[n - 3] < target) continue
    for (let i = j + 1; i < n - 2; i++) {
      if (i > j + 1 && nums[i] == nums[i - 1]) continue
      if (nums[j] + nums[i] + nums[i + 1] + nums[i + 2] > target) break
      if (nums[j] + nums[i] + nums[n - 1] + nums[n - 2] < target) continue
      let left = i + 1
      let right = n - 1
      while (left < right) {
        let sum = nums[j] + nums[i] + nums[left] + nums[right]
        if (sum > target) {
          right--
        } else if (sum < target) {
          left++
        } else {
          ans.push([nums[j], nums[i], nums[left], nums[right]])
          right--
          while (nums[right] == nums[right + 1]) right--
          left++
          while (nums[left] == nums[left - 1]) left++
        }
      }
    }
  }
  return ans
}
```
