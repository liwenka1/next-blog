[771. 宝石与石头](https://leetcode.cn/problems/jewels-and-stones/)

```js
/**
 * @param {string} jewels
 * @param {string} stones
 * @return {number}
 */
var numJewelsInStones = function (jewels, stones) {
  const set = new Set(jewels)
  let ans = 0
  for (let stone of stones) {
    if (set.has(stone)) ans++
  }
  return ans
}
```
