[1262. 可被三整除的最大和](https://leetcode.cn/problems/greatest-sum-divisible-by-three/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSumDivThree = function (nums) {
  const n = nums.length
  const f = new Array(2).fill([]).map(() => new Array(3).fill(-Infinity))
  f[0][0] = 0
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < 3; j++) {
      f[(i + 1) % 2][j] = Math.max(
        f[i % 2][j],
        f[i % 2][(j + nums[i]) % 3] + nums[i]
      )
    }
  }
  return f[n % 2][0]
}
```

[面试题 16.19. 水域大小](https://leetcode.cn/problems/pond-sizes-lcci/)

```js
/**
 * @param {number[][]} land
 * @return {number[]}
 */
var pondSizes = function (land) {
  const n = land.length
  const m = land[0].length
  const ans = []
  const dfs = (i, j) => {
    land[i][j] = 1
    let cnt = 1
    for (let k = i - 1; k <= i + 1; k++) {
      for (let t = j - 1; t <= j + 1; t++) {
        if (k >= 0 && k < n && t >= 0 && t < m && land[k][t] == 0) {
          cnt += dfs(k, t)
        }
      }
    }
    return cnt
  }
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < m; j++) {
      if (land[i][j] == 0) {
        ans.push(dfs(i, j))
      }
    }
  }
  return ans.sort((a, b) => a - b)
}
```

[2496. 数组中字符串的最大值](https://leetcode.cn/problems/maximum-value-of-a-string-in-an-array/)

```js
/**
 * @param {string[]} strs
 * @return {number}
 */
var maximumValue = function (strs) {
  let ans = 0
  for (let i = 0; i < strs.length; i++) {
    if (isNaN(strs[i])) {
      ans = Math.max(ans, strs[i].length)
    } else {
      ans = Math.max(ans, Number(strs[i]))
    }
  }
  return ans
}
```
