```js
/**
 * @param {string} s
 * @param {number[]} indices
 * @param {string[]} sources
 * @param {string[]} targets
 * @return {string}
 */
var findReplaceString = function (s, indices, sources, targets) {
  let n = s.length
  let ans = ''
  for (let i = 0; i < n; i++) {
    if (indices.includes(i)) {
      let index = indices.indexOf(i)
      let len = sources[index].length
      let str = s.slice(i, i + len)
      if (str == sources[index]) {
        ans += targets[index]
        i += len - 1
      } else ans += s[i]
    } else ans += s[i]
  }
  return ans
}
```

```js
/**
 * @param {number[][]} img1
 * @param {number[][]} img2
 * @return {number}
 */
var largestOverlap = function (img1, img2) {
  let map = new Map()
  for (let i = 0; i < img1.length; i++) {
    for (let x = 0; x < img1[i].length; x++) {
      if (img1[i][x]) {
        for (let j = 0; j < img2.length; j++) {
          for (let y = 0; y < img1[j].length; y++) {
            if (img2[j][y])
              map.set(
                i - j + '-' + (x - y),
                (map.get(i - j + '-' + (x - y)) || 0) + 1
              )
          }
        }
      }
    }
  }
  return map.size ? Math.max(...map.values()) : 0
}
```

```js
/**
 * @param {number[]} nums
 * @param {number} lower
 * @param {number} upper
 * @return {number}
 */
var countFairPairs = function (nums, lower, upper) {
  nums.sort((a, b) => a - b)
  let ans = 0
  for (let i = 0; i < nums.length; i++) {
    let left = lower_upper(nums, i, lower - nums[i])
    let right = lower_upper(nums, i, upper - nums[i] + 1)
    ans += right - left
  }
  return ans
}

const lower_upper = (nums, right, target) => {
  let left = 0
  while (left < right) {
    let mid = Math.floor((left + right) / 2)
    if (nums[mid] >= target) right = mid
    else left = mid + 1
  }
  return left
}
```

[2564. 子字符串异或查询](https://leetcode.cn/problems/substring-xor-queries/)

```js
/**
 * @param {string} s
 * @param {number[][]} queries
 * @return {number[][]}
 */
var substringXorQueries = function (s, queries) {
  let m = {}
  let i = s.indexOf('0')
  if (i >= 0) m[0] = [i, i]
  for (let l = 0, n = s.length; l < n; l++) {
    if (s[l] == '0') continue
    for (let r = l, x = 0; r < Math.min(l + 30, n); r++) {
      x = (x << 1) | (s[r].charCodeAt() & 1)
      if (!m[x]) m[x] = [l, r]
    }
  }
  let ans = []
  for (let [x, y] of queries) {
    ans.push(m[x ^ y] || [-1, -1])
  }
  return ans
}
```

```js
  let right = 0
```

```js
/**
 * @param {number[]} cardPoints
 * @param {number} k
 * @return {number}
 */
var maxScore = function (cardPoints, k) {
  let sum = 0
  for (let i = 0; i < k; i++) {
    sum += cardPoints[i]
  }
  let ans = sum
  let n = cardPoints.length
  while (k > 0) {
    k--
    n--
    sum -= cardPoints[k]
    sum += cardPoints[n]
    ans = Math.max(ans, sum)
  }
  return ans
}
```

```js
/**
 * @param {number[][]} nums
 * @return {number[]}
 */
var findDiagonalOrder = function (nums) {
  let n = nums.length
  let map = {}
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < nums[i].length; j++) {
      if (map[i + j]) {
        map[i + j].push(nums[i][j])
      } else {
        map[i + j] = [nums[i][j]]
      }
    }
  }
  let ans = []
  for (let val of Object.values(map)) {
    ans.push(...val.reverse())
  }
  return ans
}
```

```js
  let ans = 0
```

```js
/**
 * @param {number[]} nums

 */
var minOperations = function (nums) {
  let ans = 0
  let maxn = 0
  for (let num of nums) {
    maxn = Math.max(num, maxn)
    while (num != 0) {
      if (num & 1) {
        ans++
      }
      num >>= 1
    }
  }
  while (maxn != 0) {
    ans++
    maxn >>= 1
  }
```

[6472. 查询后矩阵的和](https://leetcode.cn/problems/sum-of-matrix-after-queries/)

```js
/**
 * @param {number} n
 * @param {number[][]} queries
 * @return {number}
 */
var matrixSumQueries = function (n, queries) {
  let row = new Set()
  let col = new Set()
  let ans = 0
  queries.reverse()
  for (let query of queries) {
    if (query[0] == 0 && !row.has(query[1])) {
      row.add(query[1])
      ans += (n - col.size) * query[2]
    } else if (query[0] == 1 && !col.has(query[1])) {
      col.add(query[1])
      ans += (n - row.size) * query[2]
    }
  }
  return ans
}
```
