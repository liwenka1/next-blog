[2485. 找出中枢整数](https://leetcode.cn/problems/find-the-pivot-integer/)

```js
/**
 * @param {number} n
 * @return {number}
 */
var pivotInteger = function (n) {
  let sum = 0
  for (let i = 1; i <= n; i++) {
    sum += i
  }
  let leftSum = 0
  for (let i = 1; i <= n; i++) {
    if (leftSum * 2 + i == sum) return i
    leftSum += i
  }
  return -1
}
```
