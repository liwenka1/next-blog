[2089. 找出数组排序后的目标下标](https://leetcode.cn/problems/find-target-indices-after-sorting-array/)

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var targetIndices = function (nums, target) {
  nums.sort((a, b) => a - b)
  let ans = []
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] == target) ans.push(i)
  }
  return ans
}
```

[2090. 半径为 k 的子数组平均值](https://leetcode.cn/problems/k-radius-subarray-averages/)

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var getAverages = function (nums, k) {
  let n = nums.length
  let sum = 0
  let ans = new Array(n).fill(-1)
  for (let i = 0; i < 2 * k + 1; i++) {
    sum += nums[i]
  }
  for (let i = 2 * k, j = 0; i < n; i++, j++) {
    ans[i - k] = parseInt(sum / (2 * k + 1))
    sum += nums[i + 1]
    sum -= nums[j]
  }
  return ans
}
```

[2091. 从数组中移除最大值和最小值](https://leetcode.cn/problems/removing-minimum-and-maximum-from-array/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var minimumDeletions = function (nums) {
  let n = nums.length
  let left = nums.indexOf(Math.min(...nums))
  let right = nums.indexOf(Math.max(...nums))
  let min = Math.min(left, right)
  let max = Math.max(left, right)
  return Math.min(max + 1, n - min, n - max + min + 1)
}
```

```js
/**
 * @param {string} word
 * @return {number}
 */
var minTimeToType = function (word) {
  let n = word.length
  let ans = 0
  for (let i = 1; i < n; i++) {
    ans += Math.min(
      Math.abs(word[i].charCodeAt() - word[i - 1].charCodeAt()),
      26 - Math.abs(word[i].charCodeAt() - word[i - 1].charCodeAt())
    )
  }
  ans +=
    n +
    Math.min(
      Math.abs(word[0].charCodeAt() - 'a'.charCodeAt()),
      26 - Math.abs(word[0].charCodeAt() - 'a'.charCodeAt())
    )
  return ans
}
```

[1975. 最大方阵和](https://leetcode.cn/problems/maximum-matrix-sum/)

```js
/**
 * @param {number[][]} matrix
 * @return {number}
 */
var maxMatrixSum = function (matrix) {
  let min = Infinity
  let ans = 0
  let cnt = 0
  let n = matrix.length
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      const n = Math.abs(matrix[i][j])
      min = Math.min(min, n)
      ans += n
      if (matrix[i][j] < 0) cnt ^= 1
    }
  }
  return cnt ? ans - 2 * min : ans
}
```

[1636. 按照频率将数组升序排序](https://leetcode.cn/problems/sort-array-by-increasing-frequency/)

```js
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var frequencySort = function (nums) {
  let map = {}
  for (let num of nums) {
    if (!map[num]) map[num] = 0
    map[num]++
  }
  return nums.sort((a, b) => {
    if (map[a] == map[b]) return b - a
    return map[a] - map[b]
  })
}
```

[1637. 两点之间不包含任何点的最宽垂直区域](https://leetcode.cn/problems/widest-vertical-area-between-two-points-containing-no-points/)

```js
/**
 * @param {number[][]} points
 * @return {number}
 */
var maxWidthOfVerticalArea = function (points) {
  points.sort((a, b) => a[0] - b[0])
  let ans = 0
  for (let i = 1; i < points.length; i++) {
    ans = Math.max(points[i][0] - points[i - 1][0], ans)
  }
  return ans
}
```

[1638. 统计只差一个字符的子串数目](https://leetcode.cn/problems/count-substrings-that-differ-by-one-character/)

```js
/**
 * @param {string} s
 * @param {string} t
 * @return {number}
 */
var countSubstrings = function (s, t) {
  let m = s.length
  let n = t.length
  let ans = 0
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      let temp = 0
      for (let k = 0; k + i < m && k + j < n; k++) {
        temp += s[k + i] == t[k + j] ? 0 : 1
        if (temp == 1) ans++
        else if (temp > 1) break
      }
    }
  }
  return ans
}
```

[2614. 对角线上的质数](https://leetcode.cn/problems/prime-in-diagonal/)

```js
/**
 * @param {number[][]} nums
 * @return {number}
 */
var diagonalPrime = function (nums) {
  let n = nums.length
  let ans = 0
  for (let i = 0; i < n; i++) {
    let x = nums[i][i]
    if (isPrime(x)) ans = Math.max(ans, x)
    let y = nums[i][n - 1 - i]
    if (isPrime(y)) ans = Math.max(ans, y)
  }
  return ans
}
const isPrime = (n) => {
  for (let i = 2; i * i <= n; i++) {
    if (n % i == 0) return false
  }
  return n >= 2
}
```
