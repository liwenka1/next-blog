/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var getAverages = function (nums, k) {
  let n = nums.length
  let sum = 0
  let ans = new Array(n).fill(-1)
  for (let i = 0; i < 2 * k + 1; i++) {
    sum += nums[i]
  }
  for (let i = 2 * k, j = 0; i < n; i++, j++) {
    ans[i - k] = Math.floor(sum / (2 * k + 1))
    sum += nums[i + 1]
    sum -= nums[j]
  }
  return ans
}

 * @param {number[][]} matrix

 */
var maxMatrixSum = function (matrix) {
  let min = Infinity
  let ans = 0
  let cnt = 0
  let n = matrix.length
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      const n = Math.abs(matrix[i][j])
      min = Math.min(min, n)
      ans += n

    }
  }

var frequencySort = function (nums) {
  let map = {}
  for (let num of nums) {
    if (!map[num]) map[num] = 0
    map[num]++
  }
  return nums.sort((a, b) => {
    if (map[a] == map[b]) return b - a
    return map[a] - map[b]

var diagonalPrime = function (nums) {
  let n = nums.length
  let ans = 0
  for (let i = 0; i < n; i++) {
    let x = nums[i][i]
    if (isPrime(x)) ans = Math.max(ans, x)

    if (isPrime(y)) ans = Math.max(ans, y)

  }

var distance = function (nums) {

  const n = nums.length
  for (let i = 0; i < n; i++) {
    if (!map[nums[i]]) map[nums[i]] = []

  const ans = new Array(n).fill(0)
  for (let x of Object.values(map)) {

    }
  }

/**
 * @param {number[]} nums
 * @param {number} p
 * @return {number}
 */
var minimizeMax = function (nums, p) {
  nums.sort((a, b) => a - b)
  let n = nums.length
  let left = 0
  let right = nums[n - 1] - nums[0]
  while (left < right) {
    let mid = Math.floor((left + right) / 2)
    let cnt = 0
    for (let i = 1; i < n; i++) {
      if (nums[i] - nums[i - 1] <= mid) {
        cnt++
        i++
      }
    }
    if (cnt >= p) right = mid
    else left = mid + 1
  }
  return left
}

/**
 * @param {number[][]} items
 * @param {number[]} queries
 * @return {number[]}
 */
var maximumBeauty = function (items, queries) {
  items.sort((a, b) => a[0] - b[0])
  let n = items.length
  for (let i = 1; i < n; i++) {
    items[i][1] = Math.max(items[i - 1][1], items[i][1])
  }
  const f = (x) => {
    let left = 0
    let right = n
    while (left < right) {
      let mid = Math.floor((left + right) / 2)
      if (items[mid][0] > x) right = mid
      else left = mid + 1
    }
    return left - 1 >= 0 ? items[left - 1][1] : 0
  }
  return queries.map(f)
}

 * @param {number[]} digits

var findEvenNumbers = function (digits) {
  let n = digits.length
  let ans = new Set()
  for (let i = 0; i < n; i++) {
    if (digits[i] % 2 == 0) {
      for (let j = 0; j < n; j++) {
        for (let k = 0; k < n; k++) {
          if (k != j && j != i && k != i && digits[k] != 0) {
            ans.add('' + digits[k] + digits[j] + digits[i])
          }
        }
      }
    }
  }
  return Array.from(ans).sort((a, b) => a - b)
}

/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteMiddle = function (head) {
  if (head == null || head.next == null) return null
  let slow = head
  let fast = head
  while (fast && fast.next) {
    fast = fast.next.next
    if (fast == null || fast.next == null) {
      slow.next = slow.next.next
    }
    slow = slow.next
  }
  return head
}

/**
 * @param {string} s
 * @return {number}
 */
var longestSemiRepetitiveSubstring = function (s) {
  let ans = 1
  let n = s.length
  let left = 0
  let cnt = 0
  for (let right = 1; right < n; right++) {
    cnt += s[right] == s[right - 1] ? 1 : 0
    if (cnt > 1) {
      left++
      while (s[left] != s[left - 1]) left++
      cnt--
    }
    ans = Math.max(ans, right - left + 1)
  }
  return ans
}

var smallestString = function (s) {
  s = s.split('')
  const n = s.length
  for (let i = 0; i < n; i++) {
    if (s[i] != 'a') {
      for (let j = i; j < n; j++) {
