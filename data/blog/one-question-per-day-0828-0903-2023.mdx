[57. 插入区间](https://leetcode.cn/problems/insert-interval/description/)

```java
class Solution {
    public int[][] insert(int[][] intervals, int[] newInterval) {
        int left = newInterval[0];
        int right = newInterval[1];
        boolean placed = false;
        List<int[]> ans = new ArrayList<>();
        for (int[] interval : intervals) {
            if (interval[0] > right) {
                if (!placed) {
                    ans.add(new int[]{left, right});
                    placed = true;
                }
                ans.add(interval);
            } else if (interval[1] < left) {
                ans.add(interval);
            } else {
                left = Math.min(left, interval[0]);
                right = Math.max(right, interval[1]);
            }
        }
        if (!placed) {
            ans.add(new int[]{left, right});
        }
        return  ans.toArray(new int[ans.size()][]);
    }
}
```

[823. 带因子的二叉树](https://leetcode.cn/problems/binary-trees-with-factors/description/)

```js
/**
 * @param {number[]} arr
 * @return {number}
 */
var numFactoredBinaryTrees = function (arr) {
  const MOD = 1e9 + 7
  const n = arr.length
  arr.sort((a, b) => a - b)
  const map = new Map()
  for (let i = 0; i < n; i++) {
    map.set(arr[i], i)
  }
  const memo = new Array(n).fill(-1)
  const dfs = (i) => {
    if (memo[i] != -1) return memo[i]
    let res = 1
    const val = arr[i]
    for (let j = 0; j < i; j++) {
      const x = arr[j]
      if (val % x == 0 && map.has(val / x)) {
        res += dfs(j) * dfs(map.get(val / x))
      }
    }
    return (memo[i] = res % MOD)
  }
  let ans = 0
  for (let i = 0; i < n; i++) {
    ans += dfs(i)
  }
  return ans % MOD
}
```
