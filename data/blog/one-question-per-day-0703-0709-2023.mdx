[445. 两数相加 II](https://leetcode.cn/problems/add-two-numbers-ii/)

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function (l1, l2) {
  l1 = reverseList(l1)
  l2 = reverseList(l2)
  let head = (tail = new ListNode())
  let cnt = 0
  while (l1 || l2) {
    const n1 = l1 ? l1.val : 0
    const n2 = l2 ? l2.val : 0
    const sum = n1 + n2 + cnt
    tail.next = new ListNode(sum % 10)
    tail = tail.next
    cnt = Math.floor(sum / 10)
    if (l1) {
      l1 = l1.next
    }
    if (l2) {
      l2 = l2.next
    }
  }
  if (cnt > 0) {
    tail.next = new ListNode(cnt)
  }
  return reverseList(head.next)
}

const reverseList = (head) => {
  if (head == null || head.next == null) {
    return head
  }
  const newHead = reverseList(head.next)
  head.next.next = head
  head.next = null
  return newHead
}
```

[2679. 矩阵中的和](https://leetcode.cn/problems/sum-in-a-matrix/)

```js
/**
 * @param {number[][]} nums
 * @return {number}
 */
var matrixSum = function (nums) {
  for (let num of nums) {
    num.sort((a, b) => a - b)
  }
  const m = nums.length
  const n = nums[0].length
  let ans = 0
  for (let i = 0; i < n; i++) {
    let max = -Infinity
    for (let j = 0; j < m; j++) {
      max = Math.max(nums[j][i], max)
    }
    ans += max
  }
  return ans
}
```

[2600. K 件物品的最大和](https://leetcode.cn/problems/k-items-with-the-maximum-sum/)

```js
/**
 * @param {number} numOnes
 * @param {number} numZeros
 * @param {number} numNegOnes
 * @param {number} k
 * @return {number}
 */
var kItemsWithMaximumSum = function (numOnes, numZeros, numNegOnes, k) {
  if (k <= numOnes) {
    return k
  } else if (numOnes + numZeros >= k) {
    return numOnes
  } else {
    return numOnes - (k - numOnes - numZeros)
  }
}
```

[2178. 拆分成最多数目的正偶数之和](https://leetcode.cn/problems/maximum-split-of-positive-even-integers/)

```js
/**
 * @param {number} finalSum
 * @return {number[]}
 */
var maximumEvenSplit = function (finalSum) {
  const ans = []
  if (finalSum % 2 != 0) return ans
  for (let i = 2; i <= finalSum; i += 2) {
    ans.push(i)
    finalSum -= i
  }
  ans[ans.length - 1] += finalSum
  return ans
}
```
