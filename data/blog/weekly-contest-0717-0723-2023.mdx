[2784. 检查数组是否是好的](https://leetcode.cn/problems/check-if-array-is-good/)

```js
/**
 * @param {number[]} nums
 * @return {boolean}
 */
var isGood = function (nums) {
  const max = Math.max(...nums)
  const n = nums.length
  if (max != n - 1) return false
  nums.sort((a, b) => a - b)
  for (let i = 0; i < n - 1; i++) {
    if (nums[i] != i + 1) return false
  }
  return true
}
```

[6926. 将字符串中的元音字母排序](https://leetcode.cn/problems/sort-vowels-in-a-string/)

```js
/**
 * @param {string} s
 * @return {string}
 */
var sortVowels = function (s) {
  s = s.split('')
  const path = s.filter((i) => /[aeiouAEIOU]/.test(i)).sort()
  const n = s.length
  for (let i = 0, j = 0; i < n; i++) {
    if (/[aeiouAEIOU]/.test(s[i])) {
      s[i] = path[j]
      j++
    }
  }
  return s.join('')
}
```

[6921. 按分隔符拆分字符串](https://leetcode.cn/problems/split-strings-by-separator/)

```js
/**
 * @param {string[]} words
 * @param {character} separator
 * @return {string[]}
 */
var splitWordsBySeparator = function (words, separator) {
  return words
    .map((word) => word.split(separator))
    .flat()
    .filter((i) => i != '')
}
```

[2789. 合并后数组中的最大元素](https://leetcode.cn/problems/largest-element-in-an-array-after-merge-operations/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxArrayValue = function (nums) {
  let ans = nums.pop()
  while (nums.length) {
    const pop = nums.pop()
    ans = ans >= pop ? ans + pop : pop
  }
  return ans
}
```
