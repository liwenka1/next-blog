[415. 字符串相加](https://leetcode.cn/problems/add-strings/)

```js
/**
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */
var addStrings = function (num1, num2) {
  if (num1.length < num2.length) [num1, num2] = [num2, num1]
  let diff = 0
  let ans = ''
  let i = num1.length - 1
  let j = num2.length - 1
  while (i >= 0 && j >= 0) {
    let n1 = num1[i] ? num1[i] : 0
    let n2 = num2[j] ? num2[j] : 0
    let sum = Number(n1) + Number(n2) + diff
    ans = (sum % 10) + ans
    diff = Math.floor(sum / 10)
    i--
    j--
  }
  while (i >= 0) {
    let n1 = num1[i] ? num1[i] : 0
    let sum = Number(n1) + diff
    ans = (sum % 10) + ans
    diff = Math.floor(sum / 10)
    i--
  }
  if (diff == 1) ans = diff + ans
  return ans
}
```

[874. 模拟行走机器人](https://leetcode.cn/problems/walking-robot-simulation/)

```js
/**
 * @param {number[]} commands
 * @param {number[][]} obstacles
 * @return {number}
 */
var robotSim = function (commands, obstacles) {
  const dx = [0, -1, 0, 1]
  const dy = [-1, 0, 1, 0]
  let x = 0,
    y = 0,
    dir = 2
  let ans = 0
  const set = new Set()
  for (let ob of obstacles) {
    set.add(ob.join('-'))
  }
  for (let c of commands) {
    if (c == -2) {
      dir = (dir + 3) % 4
    } else if (c == -1) {
      dir = (dir + 1) % 4
    } else {
      for (let i = 0; i < c; i++) {
        const nx = x + dx[dir]
        const ny = y + dy[dir]
        if (set.has([nx, ny].join('-'))) break
        x = nx
        y = ny
        ans = Math.max(ans, x ** 2 + y ** 2)
      }
    }
  }
  return ans
}
```

[918. 环形子数组的最大和](https://leetcode.cn/problems/maximum-sum-circular-subarray/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubarraySumCircular = function (nums) {
  const n = nums.length
  let sum = (max = min = maxSum = minSum = nums[0])
  for (let i = 1; i < n; i++) {
    sum += nums[i]
    min = Math.min(nums[i], min + nums[i])
    minSum = Math.min(minSum, min)
    max = Math.max(nums[i], max + nums[i])
    maxSum = Math.max(maxSum, max)
  }
  return maxSum < 0 ? maxSum : Math.max(maxSum, sum - minSum)
}
```

[1499. 满足不等式的最大值](https://leetcode.cn/problems/max-value-of-equation/)

```js
/**
 * @param {number[][]} points
 * @param {number} k
 * @return {number}
 */
var findMaxValueOfEquation = function (points, k) {
  const queue = []
  let ans = -Infinity
  for (let [x, y] of points) {
    while (queue.length && queue[0][0] < x - k) {
      queue.shift()
    }
    if (queue.length) {
      ans = Math.max(ans, x + y + queue[0][1])
    }
    while (queue.length && queue[queue.length - 1][1] <= y - x) {
      queue.pop()
    }
    queue.push([x, y - x])
  }
  return ans
}
```
