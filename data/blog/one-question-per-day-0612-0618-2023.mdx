[2475. 数组中不等三元组的数目](https://leetcode.cn/problems/number-of-unequal-triplets-in-array/submissions/)

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var unequalTriplets = function (nums) {
  const map = {}
  const n = nums.length
  let cnt = 0
  let ans = 0
  for (let num of nums) {
    if (!map[num]) map[num] = 0
    map[num]++
  }
  for (let key in map) {
    ans += map[key] * cnt * (n - cnt - map[key])
    cnt += map[key]
  }
  return ans
}
```

[1375. 二进制字符串前缀一致的次数](https://leetcode.cn/problems/number-of-times-binary-string-is-prefix-aligned/)

```js
/**
 * @param {number[]} flips
 * @return {number}
 */
var numTimesAllBlue = function (flips) {
  let max = 0
  let ans = 0
  const n = flips.length
  for (let i = 0; i < n; i++) {
    max = Math.max(max, flips[i])
    if (max == i + 1) ans++
  }
  return ans
}
```

[1177. 构建回文串检测](https://leetcode.cn/problems/can-make-palindrome-from-substring/)

```js
/**
 * @param {string} s
 * @param {number[][]} queries
 * @return {boolean[]}
 */
var canMakePaliQueries = function (s, queries) {
  const n = s.length
  const sum = new Array(n + 1).fill([]).map(() => new Array(26).fill(0))
  for (let i = 0; i < n; i++) {
    sum[i + 1] = [...sum[i]]
    sum[i + 1][s.charCodeAt(i) - 'a'.charCodeAt()]++
  }
  const ans = []
  for (let [left, right, k] of queries) {
    let m = 0
    for (let i = 0; i < 26; i++) {
      m += (sum[right + 1][i] - sum[left][i]) % 2
    }
    ans.push(Math.floor(m / 2) <= k)
  }
  return ans
}
```
